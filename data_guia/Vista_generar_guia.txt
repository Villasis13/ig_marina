<script>
    document.addEventListener('DOMContentLoaded', function () {
        const input = document.getElementById('id_cliente_referencia');
        const listaClientes = document.getElementById('lista_clientes_referenciar');

        // Función para mostrar u ocultar la lista
        function toggleList(display) {
            listaClientes.style.display = display ? 'block' : 'none';
        }

        // Mostrar la lista cuando el input recibe el foco
        input.addEventListener('focus', function() {
            toggleList(true);  // Mostrar la lista cuando el input recibe foco
        });

        // Ocultar la lista cuando el input pierde el foco
        input.addEventListener('blur', function() {
            // Usamos un pequeño retraso para permitir que el evento de clic en la lista se procese primero
            setTimeout(() => {
                toggleList(false);  // Ocultar la lista cuando el input pierde el foco
            }, 200); // 200ms de retraso
        });

        // Ocultar la lista cuando se selecciona un elemento de la lista
        listaClientes.addEventListener('click', function(event) {
            if (event.target.classList.contains('cliente-sugerido')) {
                toggleList(false);  // Ocultar la lista después de seleccionar un cliente
            }
        });
    });
    $wire.on('hideModal', () => {
        $('#modalAgregarCliente').modal('hide');
    });
</script>
@endscript
